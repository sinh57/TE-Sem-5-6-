# Take user input
string = input("Enter String: ")

# Display the original string
print("\nOriginal string:", string)

# Perform bitwise AND with 127 and display detailed results
print("\n--- Bitwise AND with 127 ---")
for char in string:
    # Get ASCII value and perform bitwise AND with 127
    result = ord(char) & 127
    # Display character, its ASCII, AND result, and resulting character
    print(f"'{char}' -> ASCII: {ord(char)} & 127 = {result} (char: '{chr(result)}')")

# Perform bitwise XOR with 127 and display detailed results
print("\n--- Bitwise XOR with 127 ---")
for char in string:
    # Get ASCII value and perform bitwise XOR with 127
    result = ord(char) ^ 127
    # Display character, its ASCII, XOR result, and resulting character
    print(f"'{char}' -> ASCII: {ord(char)} ^ 127 = {result} (char: '{chr(result)}')")

# Program to AND and XOR each character in a string with 127 and display the result

# Input the string from the user
string = input("Enter String: ")

# Initialize result strings for AND and XOR operations
result_and = ""
result_xor = ""

# Iterate through each character in the string
for char in string:
    # Bitwise AND with 127
    and_val = ord(char) & 127
    # Bitwise XOR with 127
    xor_val = ord(char) ^ 127

    # Append the results to the corresponding result strings
    result_and += chr(and_val)
    result_xor += chr(xor_val)

# Display the results
print("Original string: ", string)
print("AND result: ", result_and)
print("XOR result: ", result_xor)
