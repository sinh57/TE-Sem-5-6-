# Step 1: Import necessary libraries
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
from seaborn import load_dataset

# Enable inline plotting
%matplotlib inline
# Load the inbuilt titanic dataset
titanic = sns.load_dataset('titanic')

print(titanic.head(5))
#OR

# Step 3: Box plot for age distribution by gender and survival
plt.figure(figsize=(8,6))
sns.boxplot(x='sex', y='age', hue='survived', data= titanic)
plt.title('Age Distribution by Sex and Survival')
plt.xlabel('Gender')
plt.ylabel('Age')
plt.legend(title='Survived', loc='upper right', labels=['No', 'Yes'])
plt.show()




# Observations:
print("""
Observations:
- The median age of both males and females who survived is higher than those who did not survive.
- Younger males had a lower survival rate compared to younger females.
- The age distribution for females who survived is slightly higher than those who did not.
- There are more outliers among males, especially those who did not survive.
""")
