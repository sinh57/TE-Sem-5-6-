#2. Histogram for each feature
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# Load the Iris dataset
df = pd.read_csv(r"C:\Users\lenovo\OneDrive\Pictures\dsbda\input_csvfile\Iris.csv")

# Create 2x2 subplots for numeric feature distributions
fig, axes = plt.subplots(2, 2, figsize=(16, 8))

axes[0,0].set_title("Distribution of Sepal Length (cm)")
axes[0,0].hist(df["SepalLengthCm"], bins=10, color='skyblue', edgecolor='black')

axes[0,1].set_title("Distribution of Sepal Width (cm)")
axes[0,1].hist(df["SepalWidthCm"], bins=10, color='lightgreen', edgecolor='black')

axes[1,0].set_title("Distribution of Petal Length (cm)")
axes[1,0].hist(df["PetalLengthCm"], bins=10, color='lightcoral', edgecolor='black')

axes[1,1].set_title("Distribution of Petal Width (cm)")
axes[1,1].hist(df["PetalWidthCm"], bins=10, color='gold', edgecolor='black')

plt.tight_layout()
plt.show()




#3. Boxplot for each feature
